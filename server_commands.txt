# Useful Commands for Managing the animate_all.py Script

## View the Running Script
# Check if the script is running by looking for its process ID (PID)
ps -ef | grep python3

## Check the Log Output
# View the last few lines of the log file and update in real-time
tail -f animate_all_output.log

# View the entire log file
cat animate_all_output.log

## Kill the Script
# Read the PID from the PID file
PID=$(cat animate_all_pid.txt)

# Kill the script using the PID
kill $PID

# Verify that the script has been terminated
ps -ef | grep python3
